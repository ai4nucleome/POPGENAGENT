import matplotlib.pyplot as plt
import numpy as np

# Read CV errors
cv_data = []
with open('cv_errors.txt', 'r') as f:
    for line in f:
        k, cv_error = line.strip().split('\t')
        cv_data.append((int(k), float(cv_error)))

# Sort by K value
cv_data.sort(key=lambda x: x[0])
k_values = [x[0] for x in cv_data]
cv_errors = [x[1] for x in cv_data]

# Create the plot
fig, ax = plt.subplots(figsize=(10, 6))
ax.plot(k_values, cv_errors, 'b-o', linewidth=2, markersize=8)

# Find and mark the optimal K (lowest CV error)
min_idx = np.argmin(cv_errors)
optimal_k = k_values[min_idx]
min_cv = cv_errors[min_idx]

ax.plot(optimal_k, min_cv, 'ro', markersize=12)
ax.annotate(f'Optimal K={optimal_k}\nCV={min_cv:.4f}', 
            xy=(optimal_k, min_cv), 
            xytext=(optimal_k + 0.5, min_cv + 0.002),
            arrowprops=dict(arrowstyle='->', color='red'),
            fontsize=10,
            bbox=dict(boxstyle='round,pad=0.3', facecolor='yellow', alpha=0.7))

# Formatting
ax.set_xlabel('K (Number of Ancestral Populations)', fontsize=12)
ax.set_ylabel('Cross-Validation Error', fontsize=12)
ax.set_title('ADMIXTURE Cross-Validation Error vs K', fontsize=14, fontweight='bold')
ax.grid(True, alpha=0.3)
ax.set_xticks(k_values)

# Add a box with summary statistics
textstr = f'K range: {min(k_values)}-{max(k_values)}\nOptimal K: {optimal_k}\nMin CV error: {min_cv:.4f}'
props = dict(boxstyle='round', facecolor='wheat', alpha=0.5)
ax.text(0.95, 0.95, textstr, transform=ax.transAxes, fontsize=10,
        verticalalignment='top', horizontalalignment='right', bbox=props)

plt.tight_layout()
plt.savefig('optimal_k_plot.png', dpi=300, bbox_inches='tight')
plt.close()

print(f"Optimal K value: {optimal_k} with CV error: {min_cv:.4f}")
