import numpy as np

# Function to read PLINK files and extract population information
def read_plink_data(prefix):
    # Read FAM file to get population information
    fam_file = prefix + ".fam"
    fam_data = pd.read_csv(fam_file, sep='\s+', header=None, 
                           names=['FID', 'IID', 'PID', 'MID', 'Sex', 'Pheno'])
    
    # Extract population codes from FID (assuming format like CEU_NA12878)
    fam_data['Population'] = fam_data['FID'].str.extract(r'([A-Z]{3})')
    
    # Get unique populations
    populations = fam_data['Population'].unique()
    populations = populations[pd.notna(populations)]
    
    return fam_data, populations

# Function to calculate allele frequencies for each population
def calculate_allele_frequencies(prefix, fam_data, populations):
    print("Calculating allele frequencies for each population...")
    
    allele_freq_dict = {}
    
    for pop in populations:
        # Get samples for this population
        pop_samples = fam_data[fam_data['Population'] == pop][['FID', 'IID']]
        
        # Write sample list to file
        sample_file = f"./output/019/samples_{pop}.txt"
        pop_samples.to_csv(sample_file, sep='\t', header=False, index=False)
        
        # Use PLINK to calculate allele frequencies
        cmd = f"plink --bfile {prefix} --keep {sample_file} --freq --out ./output/019/freq_{pop}"
        subprocess.run(cmd, shell=True, check=True)
        
        # Read frequency file
        freq_file = f"./output/019/freq_{pop}.frq"
        if os.path.exists(freq_file):
            freq_data = pd.read_csv(freq_file, sep='\s+')
            allele_freq_dict[pop] = freq_data
    
    return allele_freq_dict

# Function to calculate 3P-CLR statistic
def calculate_3pclr(freq1, freq2, freq3, window_size=100):
    """
    Calculate 3P-CLR statistic for three populations
    This detects selection and admixture by comparing allele frequency patterns
    """
    # Merge frequency data on chromosome and position
    merged = freq1.merge(freq2, on=['CHR', 'SNP'], suffixes=('_1', '_2'))
    merged = merged.merge(freq3, on=['CHR', 'SNP'])
    merged = merged.rename(columns={'MAF': 'MAF_3'})
    
    # Calculate frequency differences
    merged['p1'] = merged['MAF_1']
    merged['p2'] = merged['MAF_2']
    merged['p3'] = merged['MAF_3']
    
    # Remove SNPs with missing data or fixed alleles
    merged = merged[(merged['p1'] > 0) & (merged['p1'] < 1) &
                    (merged['p2'] > 0) & (merged['p2'] < 1) &
                    (merged['p3'] > 0) & (merged['p3'] < 1)]
    
    # Calculate 3P-CLR components
    # This measures deviation from neutral evolution expectations
    merged['d12'] = merged['p1'] - merged['p2']
    merged['d13'] = merged['p1'] - merged['p3']
    merged['d23'] = merged['p2'] - merged['p3']
    
    # Calculate composite likelihood ratio statistic
    # Higher values indicate selection or admixture
    merged['clr_component'] = (merged['d12']**2 + merged['d13']**2 + merged['d23']**2) / \
                              (merged['p1']*(1-merged['p1']) + merged['p2']*(1-merged['p2']) + \
                               merged['p3']*(1-merged['p3']) + 1e-6)
    
    # Calculate windowed 3P-CLR
    results = []
    for chrom in merged['CHR'].unique():
        chrom_data = merged[merged['CHR'] == chrom].copy()
        chrom_data = chrom_data.sort_values('A1_1')  # Sort by position proxy
        
        # Create windows
        n_snps = len(chrom_data)
        for i in range(0, n_snps, window_size):
            window_data = chrom_data.iloc[i:min(i+window_size, n_snps)]
            if len(window_data) > 0:
                clr_stat = np.sum(window_data['clr_component'])
                results.append({
                    'CHR': chrom,
                    'START_SNP': window_data.iloc[0]['SNP'],
                    'END_SNP': window_data.iloc[-1]['SNP'],
                    'N_SNPS': len(window_data),
                    '3P_CLR': clr_stat
                })
    
    return pd.DataFrame(results)
