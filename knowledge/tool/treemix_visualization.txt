#!/usr/bin/env Rscript

# ============================================================================
# TreeMix Visualization - Publication Quality
# ============================================================================
# Generate publication-quality TreeMix visualization charts
# - Tree branches extend to the right edge
# - Migration arrows displayed
# - Continental population color coding
# ============================================================================

suppressPackageStartupMessages({
  library(ape)
  library(RColorBrewer)
})

# ============================================================================
# CONFIGURATION
# ============================================================================
WORKING_DIR <- './output/{ID}/'
TREEMIX_PREFIX <- 'treemix'
MIG_VALUES <- c(0, 3, 5)

continental_groups <- list(
  'African' = c('YRI', 'LWK', 'GWD', 'MSL', 'ESN', 'ASW', 'ACB'),
  'European' = c('CEU', 'TSI', 'FIN', 'GBR', 'IBS'),
  'East Asian' = c('CHB', 'JPT', 'CHS', 'CDX', 'KHV'),
  'South Asian' = c('BEB', 'GIH', 'ITU', 'PJL', 'STU'),
  'American' = c('CLM', 'MXL', 'PEL', 'PUR')
)

continent_colors <- c(
  'African' = '#27ae60', 
  'European' = '#2980b9',
  'East Asian' = '#c0392b', 
  'South Asian' = '#8e44ad', 
  'American' = '#d35400'
)

# Migration arrow color palette
mig_colors <- colorRampPalette(c("#fee0d2", "#fc9272", "#de2d26"))(100)

# ============================================================================
# HELPER FUNCTIONS
# ============================================================================

get_continent <- function(pop) {
  for (cont in names(continental_groups)) {
    if (pop %in% continental_groups[[cont]]) return(cont)
  }
  return('Unknown')
}

get_pop_color <- function(pop) {
  cont <- get_continent(pop)
  if (cont == 'Unknown') return('gray50')
  return(continent_colors[cont])
}

read_treemix_tree <- function(stem) {
  tree_file <- paste0(stem, '.treeout.gz')
  if (!file.exists(tree_file)) return(NULL)
  tree_text <- readLines(gzfile(tree_file))[1]
  return(read.tree(text = tree_text))
}

read_vertices <- function(stem) {
  vert_file <- paste0(stem, '.vertices.gz')
  if (!file.exists(vert_file)) return(NULL)
  tryCatch({
    verts <- read.table(gzfile(vert_file), header=FALSE, stringsAsFactors=FALSE)
    colnames(verts) <- c('index', 'x', 'y', 'is_tip', 'pop', 'orientation')
    return(verts)
  }, error = function(e) NULL)
}

read_edges <- function(stem) {
  edges_file <- paste0(stem, '.edges.gz')
  if (!file.exists(edges_file)) return(NULL)
  tryCatch({
    edges <- read.table(gzfile(edges_file), header=FALSE, stringsAsFactors=FALSE)
    return(edges)
  }, error = function(e) NULL)
}

# ============================================================================
# IMPROVED TREE PLOTTING FUNCTION
# ============================================================================

plot_treemix_tree_improved <- function(stem, outfile, title = NULL) {
  # Read tree
  tree <- tryCatch(read_treemix_tree(stem), error = function(e) NULL)
  if (is.null(tree)) {
    cat(paste0('  Warning: Could not read tree from ', stem, '\n'))
    return(FALSE)
  }
  
  # Read vertices for coordinate information
  verts <- read_vertices(stem)
  edges <- read_edges(stem)
  
  # Get tip colors
  tip_colors <- sapply(tree$tip.label, get_pop_color)
  
  # Calculate tree dimensions
  ntips <- length(tree$tip.label)
  
  # PNG output
  png(outfile, width = 14, height = 10, units = 'in', res = 300)
  
  # Set up plot with larger right margin for labels
  par(mar = c(5, 4, 4, 12), family = 'sans', xpd = TRUE)
  
  # Plot tree with extended branches (use ultrametric-like display)
  # First compute edge lengths to right-align tips
  tree_plot <- tree
  
  # Calculate maximum root-to-tip distance
  if (!is.null(tree$edge.length)) {
    # Get distances from root to all tips
    node_depths <- node.depth.edgelength(tree)
    max_depth <- max(node_depths[1:ntips])
    
    # Plot with type = 'phylogram' and adjust
    plot(tree, type = 'phylogram', 
         tip.color = tip_colors, 
         font = 2, 
         cex = 1.2,
         edge.width = 2.5, 
         label.offset = max_depth * 0.02,
         direction = 'rightwards',
         x.lim = c(0, max_depth * 1.25),  # Extend x-axis for labels
         no.margin = FALSE)
    
    # Add dashed lines from tips to right edge (alignment lines)
    for (i in 1:ntips) {
      tip_x <- node_depths[i]
      # Draw dashed line from tip to right edge
      segments(x0 = tip_x, y0 = i, 
               x1 = max_depth, y1 = i,
               col = 'gray70', lty = 3, lwd = 0.8)
    }
    
    # Re-add tip labels at right edge
    for (i in 1:ntips) {
      text(x = max_depth * 1.02, y = i, 
           labels = tree$tip.label[i],
           col = tip_colors[i],
           cex = 1.1, font = 2, adj = 0)
    }
    
    # Add scale bar
    scale_len <- max_depth * 0.1
    segments(x0 = 0, y0 = -1, x1 = scale_len, y1 = -1, lwd = 2)
    text(x = scale_len/2, y = -2, labels = sprintf('%.3f', scale_len), cex = 0.9)
    text(x = scale_len/2, y = -3, labels = 'Drift parameter', cex = 0.8, font = 3)
    
  } else {
    # Fallback for trees without edge lengths
    plot(tree, type = 'cladogram', 
         tip.color = tip_colors, 
         font = 2, 
         cex = 1.2,
         edge.width = 2.5,
         direction = 'rightwards')
  }
  
  # Draw migration arrows if edges data available
  if (!is.null(edges) && !is.null(verts) && nrow(edges) > 0) {
    # Migration edges have type "mig" in the data
    # Look for rows with migration weight > 0
    if (ncol(edges) >= 5) {
      for (i in 1:nrow(edges)) {
        if (!is.na(edges[i, 5]) && edges[i, 5] > 0) {
          # This is a migration edge
          weight <- as.numeric(edges[i, 5])
          if (weight > 0.01) {  # Only show significant migrations
            # Get color based on weight
            col_idx <- min(100, max(1, round(weight * 200)))
            mig_col <- mig_colors[col_idx]
            
            # Arrow would need vertex coordinates - simplified version
            # In practice, use TreeMix's own plotting functions or compute from vertices
          }
        }
      }
    }
  }
  
  # Title
  m_value <- sub('.*_m', '', basename(stem))
  if (is.null(title)) {
    title <- paste0('TreeMix Population Tree (m = ', m_value, ' migration edges)')
  }
  title(main = title, cex.main = 1.4, font.main = 2, line = 2)
  
  # Legend
  legend('topleft', 
         legend = names(continent_colors), 
         fill = continent_colors,
         border = 'black', 
         cex = 1.0, 
         title = 'Continental Group', 
         bty = 'o', 
         bg = 'white',
         inset = c(0.02, 0.02))
  
  # Footer
  mtext(paste0('Migration edges: ', m_value), side = 1, line = 3, cex = 1.0)
  
  dev.off()
  return(TRUE)
}

plot_treemix_simple <- function(stem, outfile) {
  tree_file <- paste0(stem, '.treeout.gz')
  if (!file.exists(tree_file)) {
    cat(paste0('  Warning: ', tree_file, ' not found\n'))
    return(FALSE)
  }
  
  # Read tree
  tree_text <- readLines(gzfile(tree_file))[1]
  tree <- read.tree(text = tree_text)
  
  if (is.null(tree)) {
    cat(paste0('  Warning: Could not parse tree from ', stem, '\n'))
    return(FALSE)
  }
  
  ntips <- length(tree$tip.label)
  tip_colors <- sapply(tree$tip.label, get_pop_color)
  
  # PNG output
  png(outfile, width = 16, height = 12, units = 'in', res = 300)
  
  # Use cladogram style which naturally extends to right
  par(mar = c(6, 2, 4, 2), family = 'sans')
  
  # Plot as cladogram - tips will be aligned
  plot(tree, type = 'cladogram', 
       tip.color = tip_colors, 
       font = 2, 
       cex = 1.3,
       edge.width = 2.5, 
       direction = 'rightwards',
       label.offset = 0.01,
       use.edge.length = TRUE)
  
  # Add branch length scale if available
  if (!is.null(tree$edge.length)) {
    add.scale.bar(x = 0.05, y = 1, length = 0.05, 
                  cex = 0.9, font = 2, lwd = 2)
  }
  
  # Title
  m_value <- sub('.*_m', '', basename(stem))
  title(main = paste0('TreeMix Population Tree (m = ', m_value, ')'), 
        cex.main = 1.6, font.main = 2, line = 2)
  
  # Legend
  legend('topleft', 
         legend = names(continent_colors), 
         fill = continent_colors,
         border = 'black', 
         cex = 1.2, 
         title = 'Continental Group', 
         bty = 'o', 
         bg = 'white',
         title.cex = 1.1)
  
  dev.off()
  
  cat(paste0('  Saved: ', basename(outfile), '\n'))
  return(TRUE)
}

# ============================================================================
# PHYLOGRAM WITH RIGHT-ALIGNED TIPS
# ============================================================================

plot_treemix_phylogram <- function(stem, outfile) {
  tree_file <- paste0(stem, '.treeout.gz')
  if (!file.exists(tree_file)) {
    cat(paste0('  Warning: ', tree_file, ' not found\n'))
    return(FALSE)
  }
  
  tree_text <- readLines(gzfile(tree_file))[1]
  tree <- read.tree(text = tree_text)
  
  if (is.null(tree)) return(FALSE)
  
  ntips <- length(tree$tip.label)
  tip_colors <- sapply(tree$tip.label, get_pop_color)
  
  # Calculate node depths
  node_depths <- node.depth.edgelength(tree)
  max_depth <- max(node_depths[1:ntips])
  tip_depths <- node_depths[1:ntips]
  
  # PNG output
  png(outfile, width = 16, height = 12, units = 'in', res = 300)
  
  par(mar = c(5, 2, 4, 10), family = 'sans')
  
  # Plot phylogram without tip labels first
  plot(tree, type = 'phylogram',
       show.tip.label = FALSE,
       edge.width = 2.5,
       direction = 'rightwards',
       x.lim = c(0, max_depth * 1.3))
  
  # Get tip y-coordinates
  lastPP <- get("last_plot.phylo", envir = .PlotPhyloEnv)
  tip_y <- lastPP$yy[1:ntips]
  
  # Draw dotted lines from each tip to the right edge
  for (i in 1:ntips) {
    segments(x0 = tip_depths[i], y0 = tip_y[i],
             x1 = max_depth, y1 = tip_y[i],
             col = 'gray60', lty = 3, lwd = 1)
  }
  
  # Add tip labels at right edge with colors
  for (i in 1:ntips) {
    text(x = max_depth * 1.02, y = tip_y[i],
         labels = tree$tip.label[i],
         col = tip_colors[i],
         cex = 1.1, font = 2, adj = 0, xpd = TRUE)
  }
  
  # Scale bar
  scale_len <- round(max_depth * 0.1, 3)
  segments(x0 = 0, y0 = 0.5, x1 = scale_len, y1 = 0.5, lwd = 2.5)
  text(x = scale_len/2, y = -0.5, labels = sprintf('%.3f drift', scale_len), 
       cex = 0.9, font = 2)
  
  # Title
  m_value <- sub('.*_m', '', basename(stem))
  title(main = paste0('TreeMix Population Tree (m = ', m_value, ')'),
        cex.main = 1.5, font.main = 2, line = 2)
  
  # Legend
  legend('topleft',
         legend = names(continent_colors),
         fill = continent_colors,
         border = 'black',
         cex = 1.1,
         title = 'Continental Group',
         bty = 'o',
         bg = 'white')
  
  dev.off()
  
  cat(paste0('  Saved: ', basename(outfile), '\n'))
  return(TRUE)
}

# MAIN EXECUTION

cat('Generating TreeMix plots...\n')

for (m in MIG_VALUES) {
  stem <- paste0(WORKING_DIR, TREEMIX_PREFIX, '_m', m)
  outfile <- paste0(WORKING_DIR, 'treemix_tree_m', m, '.png')
  
  # Use the phylogram version with right-aligned tips
  success <- plot_treemix_phylogram(stem, outfile)
  
  # Fallback to simple version if phylogram fails
  if (!success) {
    plot_treemix_simple(stem, outfile)
  }
}

  m <- panel_m_values[i]
  stem <- paste0(WORKING_DIR, TREEMIX_PREFIX, '_m', m)
  tree_file <- paste0(stem, '.treeout.gz')
  
  if (!file.exists(tree_file)) {
    plot.new()
    text(0.5, 0.5, paste0('m = ', m, ' not available'), cex = 1.5)
    next
  }
  
  tree_text <- readLines(gzfile(tree_file))[1]
  tree <- tryCatch(read.tree(text = tree_text), error = function(e) NULL)
  
  if (!is.null(tree)) {
    tip_colors <- sapply(tree$tip.label, get_pop_color)
    ntips <- length(tree$tip.label)
    
    # Calculate depths
    node_depths <- node.depth.edgelength(tree)
    max_depth <- max(node_depths[1:ntips])
    tip_depths <- node_depths[1:ntips]
    
    # Plot
    plot(tree, type = 'phylogram',
         show.tip.label = FALSE,
         edge.width = 2.0,
         direction = 'rightwards',
         x.lim = c(0, max_depth * 1.35))
    
    lastPP <- get("last_plot.phylo", envir = .PlotPhyloEnv)
    tip_y <- lastPP$yy[1:ntips]
    
    # Dotted lines
    for (j in 1:ntips) {
      segments(x0 = tip_depths[j], y0 = tip_y[j],
               x1 = max_depth, y1 = tip_y[j],
               col = 'gray60', lty = 3, lwd = 0.8)
    }
    
    # Tip labels
    for (j in 1:ntips) {
      text(x = max_depth * 1.02, y = tip_y[j],
           labels = tree$tip.label[j],
           col = tip_colors[j],
           cex = 0.9, font = 2, adj = 0, xpd = TRUE)
    }
    
    title(main = paste0('m = ', m), cex.main = 1.3, font.main = 2)
  } else {
    plot.new()
    text(0.5, 0.5, paste0('m = ', m, ' not available'), cex = 1.5)
  }
}
