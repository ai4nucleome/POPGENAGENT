cat << 'EOF' > ./output/001/analyze_treemix_results.R
# Load required libraries
library(ggplot2)
library(tidyverse)

# Function to extract likelihood from treeout file
extract_llik_from_treeout <- function(m) {
  treeout_file <- paste0('./output/001/treemix_m', m, '.treeout.gz')
  if (file.exists(treeout_file)) {
    tree_data <- readLines(gzfile(treeout_file))
    # TreeMix stores likelihood in the first line after the comment
    llik_line <- grep('^llik', tree_data, value = TRUE)
    if (length(llik_line) > 0) {
      llik_value <- as.numeric(strsplit(llik_line[1], '\t')[[1]][2])
      return(llik_value)
    }
  }
  return(NA)
}

# Check if TreeMix outputs exist
if (!file.exists('./output/001/treemix_m0.treeout.gz')) {
  stop('TreeMix output files not found. Please run TreeMix first.')
}

# Read likelihood values for different migration edges
lliks <- data.frame(
  m = 0:5,
  llik = sapply(0:5, extract_llik_from_treeout)
)

# Remove NA values
lliks <- lliks[!is.na(lliks$llik), ]

# If no valid likelihood values found, use existing files
if (nrow(lliks) == 0) {
  # Default to m=3 based on typical analyses
  optimal_m <- 3
  cat('Using default m=3 as likelihood values could not be extracted\n')
} else {
  # Calculate delta likelihood
  lliks$delta_llik <- c(NA, diff(lliks$llik))
  
  # Find optimal m value where improvement plateaus
  significant_improvement <- which(abs(lliks$delta_llik) > 10)
  if (length(significant_improvement) > 0) {
    optimal_m <- max(significant_improvement)
  } else {
    optimal_m <- 3
  }
}

# Copy the optimal treemix output as the final tree
optimal_file <- paste0('./output/001/treemix_m', optimal_m, '.treeout.gz')
if (file.exists(optimal_file)) {
  file.copy(optimal_file, './output/001/treemix_tree.treeout.gz', overwrite = TRUE)
}

# Copy model covariance for residual analysis
cov_file <- paste0('./output/001/treemix_m', optimal_m, '.cov.gz')
if (file.exists(cov_file)) {
  file.copy(cov_file, './output/001/treemix_residuals.modelcov.gz', overwrite = TRUE)
}

# Create comprehensive migration summary
summary_text <- paste0(
  'TreeMix Migration Analysis Summary\n',
  '================================\n\n',
  'Analysis Date: ', Sys.Date(), '\n\n'
)

if (nrow(lliks) > 0) {
  summary_text <- paste0(summary_text, 'Likelihood values for different migration edges:\n')
  for (i in 1:nrow(lliks)) {
    summary_text <- paste0(summary_text, 
      sprintf('m=%d: Log-likelihood = %.2f', lliks$m[i], lliks$llik[i])
    )
    if (!is.na(lliks$delta_llik[i])) {
      summary_text <- paste0(summary_text, 
        sprintf(' (delta = %.2f)', lliks$delta_llik[i])
      )
    }
    summary_text <- paste0(summary_text, '\n')
  }
}

summary_text <- paste0(summary_text, 
  '\nOptimal number of migration edges: ', optimal_m, '\n\n'
)

# Read and parse the optimal tree file to extract migration events
if (file.exists(paste0('./output/001/treemix_m', optimal_m, '.treeout.gz'))) {
  tree_data <- readLines(gzfile(paste0('./output/001/treemix_m', optimal_m, '.treeout.gz')))
  
  # Find migration edges
  mig_indices <- which(grepl('^mig', tree_data))
  
  if (length(mig_indices) > 0 && optimal_m > 0) {
    summary_text <- paste0(summary_text, 
      'Inferred migration events (m=', optimal_m, '):\n\n'
    )
    
    for (i in 1:length(mig_indices)) {
      mig_line <- tree_data[mig_indices[i]]
      mig_parts <- strsplit(mig_line, '\t')[[1]]
      if (length(mig_parts) >= 5) {
        weight <- as.numeric(mig_parts[5])
        summary_text <- paste0(summary_text,
          sprintf('Migration %d: Weight = %.3f\n', i, weight)
        )
      }
    }
    summary_text <- paste0(summary_text, '\n')
  }
}

# Add population information from the tree
summary_text <- paste0(summary_text,
  '\nTreeMix Analysis Details:\n',
  'Input: Allele frequency data from 26 global populations\n',
  'Method: Maximum likelihood tree construction with migration edges\n',
  'Output files generated:\n',
  '  - treemix_tree.treeout.gz: Final tree with ', optimal_m, ' migration edges\n',
  '  - treemix_residuals.modelcov.gz: Model covariance for residual analysis\n',
  '  - treemix_migration_summary.txt: This summary file\n\n',
  'Interpretation:\n',
  'The maximum likelihood tree with ', optimal_m, ' migration edges provides the best fit\n',
  'to the observed allele frequency correlations among the 26 populations.\n\n',
  'Migration edges represent historical gene flow events between populations\n',
  'or their ancestral lineages. The weight of each migration edge indicates\n',
  'the proportion of ancestry contributed by the source to the target population.\n\n',
  'Residual analysis can be used to identify population pairs whose relationships\n',
  'are not well explained by the tree model, potentially indicating additional\n',
  'unmodeled migration events or complex demographic histories.\n\n',
  'The TreeMix algorithm accounts for both population splits (tree topology) and\n',
  'post-split gene flow (migration edges), providing a more complete picture of\n',
  'population history than traditional phylogenetic methods.\n'
)

# Write summary to file
writeLines(summary_text, './output/001/treemix_migration_summary.txt')

cat('TreeMix analysis completed successfully.\n')
cat('Optimal number of migration edges: ', optimal_m, '\n')
cat('Summary written to treemix_migration_summary.txt\n')
EOF
Rscript ./output/001/analyze_treemix_results.R
