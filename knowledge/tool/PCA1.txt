import pandas as pd
import matplotlib.pyplot as plt
import numpy as np

# Define population codes and colors for all 26 populations
pop_colors = {
    'ACB': '#FF6B6B', 'ASW': '#4ECDC4', 'BEB': '#45B7D1', 'CDX': '#96CEB4',
    'CEU': '#FECA57', 'CHB': '#FF9FF3', 'CHS': '#54A0FF', 'CLM': '#48DBFB',
    'ESN': '#0ABDE3', 'FIN': '#EE5A24', 'GBR': '#10AC84', 'GIH': '#5F27CD',
    'GWD': '#C8D6E5', 'IBS': '#576574', 'ITU': '#222F3E', 'JPT': '#F368E0',
    'KHV': '#FF6348', 'LWK': '#00D2D3', 'MSL': '#55EFC4', 'MXL': '#FDCB6E',
    'PEL': '#6C5CE7', 'PJL': '#A29BFE', 'PUR': '#FD79A8', 'STU': '#BADC58',
    'TSI': '#FF7979', 'YRI': '#686DE0'
}

# Read eigenvector file
with open('./output/019/pca_results.evec', 'r') as f:
    lines = f.readlines()

# Parse eigenvectors
data = []
for line in lines:
    if not line.startswith('#'):
        parts = line.strip().split()
        if len(parts) >= 22:  # ID + 20 PCs + control
            data.append(parts)

# Create DataFrame
columns = ['ID'] + [f'PC{i}' for i in range(1, 21)] + ['Control']
df = pd.DataFrame(data, columns=columns)

# Extract population from ID
df['Population'] = df['ID'].str.split('_').str[0]

# Convert PCs to numeric
for i in range(1, 21):
    df[f'PC{i}'] = pd.to_numeric(df[f'PC{i}'], errors='coerce')

# Read eigenvalues
with open('./output/019/pca_results.eval', 'r') as f:
    eigenvalues = [float(line.strip()) for line in f.readlines() if line.strip()]

# Calculate variance explained
total_var = sum(eigenvalues)
var_explained = [(ev/total_var)*100 for ev in eigenvalues]

# Create PC1 vs PC2 plot
plt.figure(figsize=(10, 8))
for pop in pop_colors:
    pop_data = df[df['Population'] == pop]
    if len(pop_data) > 0:
        plt.scatter(pop_data['PC1'], pop_data['PC2'], 
                   c=pop_colors[pop], label=pop, s=40, alpha=0.7, edgecolors='none')
plt.xlabel(f'PC1 ({var_explained[0]:.2f}% variance)')
plt.ylabel(f'PC2 ({var_explained[1]:.2f}% variance)')
plt.title('PCA of 26 Populations - PC1 vs PC2')
plt.legend(bbox_to_anchor=(1.05, 1), loc='upper left', ncol=2, frameon=False)
plt.tight_layout()
plt.savefig('./output/019/pca_plot_PC1_PC2.png', dpi=300, bbox_inches='tight')
plt.close()

# Create PC1 vs PC3 plot
plt.figure(figsize=(10, 8))
for pop in pop_colors:
    pop_data = df[df['Population'] == pop]
    if len(pop_data) > 0:
        plt.scatter(pop_data['PC1'], pop_data['PC3'], 
                   c=pop_colors[pop], label=pop, s=40, alpha=0.7, edgecolors='none')
plt.xlabel(f'PC1 ({var_explained[0]:.2f}% variance)')
plt.ylabel(f'PC3 ({var_explained[2]:.2f}% variance)')
plt.title('PCA of 26 Populations - PC1 vs PC3')
plt.legend(bbox_to_anchor=(1.05, 1), loc='upper left', ncol=2, frameon=False)
plt.tight_layout()
plt.savefig('./output/019/pca_plot_PC1_PC3.png', dpi=300, bbox_inches='tight')
plt.close()

# Create scree plot
plt.figure(figsize=(10, 6))
plt.bar(range(1, min(21, len(var_explained)+1)), var_explained[:20])
plt.xlabel('Principal Component')
plt.ylabel('Variance Explained (%)')
plt.title('Scree Plot - Variance Explained by Principal Components')
plt.xticks(range(1, 21))
plt.grid(axis='y', alpha=0.3)
plt.tight_layout()
plt.savefig('./output/019/pca_variance_explained.png', dpi=300)
plt.close()

print('PCA visualization complete.')
print(f'Total samples analyzed: {len(df)}')
print(f'Number of populations: {df["Population"].nunique()}')
print(f'First 5 PCs explain {sum(var_explained[:5]):.2f}% of variance')
EOF
