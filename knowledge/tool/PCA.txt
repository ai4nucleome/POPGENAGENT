import sys

# Read the FAM file to extract unique populations
populations = set()
with open("./output/030/final_ld_pruned_1000GP.fam", "r") as f:
    for line in f:
        parts = line.strip().split()
        if len(parts) >= 1:
            # Extract population code from the first part of FID (before underscore)
            pop = parts[0].split("_")[0]
            populations.add(pop)

# Sort populations and print
pop_list = sorted(list(populations))
print("Found", len(pop_list), "populations:", pop_list)

# Create a color palette for all populations
import colorsys
colors = {}
n = len(pop_list)
for i, pop in enumerate(pop_list):
    hue = i / n
    rgb = colorsys.hsv_to_rgb(hue, 0.8, 0.9)
    colors[pop] = "#{:02x}{:02x}{:02x}".format(int(rgb[0]*255), int(rgb[1]*255), int(rgb[2]*255))

# Save population info for later use
with open("./output/030/population_info.txt", "w") as f:
    f.write("POPULATIONS=" + str(pop_list) + "\n")
    f.write("COLORS=" + str(colors) + "\n")
    f.write("NUM_POPS=" + str(n) + "\n")

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import statsmodels.api as sm
import numpy as np
from sklearn.preprocessing import LabelEncoder
import pathlib
import ast

# Read population info
with open("./output/030/population_info.txt", "r") as f:
    lines = f.readlines()
    for line in lines:
        if line.startswith("POPULATIONS="):
            POPULATIONS = ast.literal_eval(line.strip().split("=", 1)[1])
        elif line.startswith("COLORS="):
            COLORS = ast.literal_eval(line.strip().split("=", 1)[1])
        elif line.startswith("NUM_POPS="):
            NUM_POPS = int(line.strip().split("=")[1])

# Read PCA results
rows = []
with open("./output/030/pca_results.txt", "r") as f:
    for line in f:
        if not line.startswith("#"):
            parts = line.strip().split()
            if parts:
                rows.append(parts)

if not rows:
    print("Error: No PCA results found")
    sys.exit(1)

# Determine the number of columns
max_cols = max(len(row) for row in rows)
for row in rows:
    row.extend(["NA"] * (max_cols - len(row)))

# Create column names
col_names = ["FID"] + [f"PC{i}" for i in range(1, max_cols-1)] + ["SEX"]
df = pd.DataFrame(rows, columns=col_names)

# Extract population from FID
df["POP"] = df["FID"].str.split("_").str[0]

# Convert PC columns to numeric
for i in range(1, 21):
    pc_col = f"PC{i}"
    if pc_col in df.columns:
        df[pc_col] = pd.to_numeric(df[pc_col], errors="coerce")

# Filter to only include known populations
df = df[df["POP"].isin(POPULATIONS)].reset_index(drop=True)

# Create numeric encoding for populations
le = LabelEncoder()
df["POP_NUM"] = le.fit_transform(df["POP"])

# Calculate variance explained by each PC
with open("./output/030/pca_eval.txt", "r") as f:
    eigenvalues = [float(line.strip()) for line in f if line.strip()]

total_var = sum(eigenvalues)
var_explained = [(ev/total_var)*100 for ev in eigenvalues[:10]]

# Create PCA visualization (PC1 vs PC2)
plt.figure(figsize=(12, 8))
for pop in POPULATIONS:
    pop_data = df[df["POP"] == pop]
    plt.scatter(pop_data["PC1"], pop_data["PC2"], 
                c=COLORS[pop], label=pop, s=30, alpha=0.7, edgecolors="none")

plt.xlabel(f"PC1 ({var_explained[0]:.2f}% variance)")
plt.ylabel(f"PC2 ({var_explained[1]:.2f}% variance)")
plt.title("Principal Component Analysis - 1000 Genomes Project (26 Populations)")
plt.legend(bbox_to_anchor=(1.05, 1), loc="upper left", ncol=2, fontsize=8)
plt.tight_layout()
plt.savefig("./output/030/pca.png", dpi=300, bbox_inches="tight")
plt.close()

# Create additional PC plots
fig, axes = plt.subplots(2, 2, figsize=(14, 12))
pc_pairs = [(1,2), (1,3), (2,3), (3,4)]

for idx, (ax, (pc1, pc2)) in enumerate(zip(axes.flatten(), pc_pairs)):
    for pop in POPULATIONS:
        pop_data = df[df["POP"] == pop]
        ax.scatter(pop_data[f"PC{pc1}"], pop_data[f"PC{pc2}"], 
                   c=COLORS[pop], label=pop if idx==0 else "", s=20, alpha=0.6)
    
    ax.set_xlabel(f"PC{pc1} ({var_explained[pc1-1]:.2f}% variance)")
    ax.set_ylabel(f"PC{pc2} ({var_explained[pc2-1]:.2f}% variance)")
    ax.set_title(f"PC{pc1} vs PC{pc2}")

if idx == 0:
    ax.legend(bbox_to_anchor=(1.05, 1), loc="upper left", ncol=1, fontsize=6)
    
fig.suptitle("Multiple PC Comparisons - 1000 Genomes Project", fontsize=14)
plt.tight_layout()
plt.savefig("./output/030/pca_multiplot.png", dpi=300, bbox_inches="tight")
plt.close()

# Regression analysis - Population as outcome
# Use first 10 PCs as predictors
pc_cols = [f"PC{i}" for i in range(1, 11) if f"PC{i}" in df.columns]
X = df[pc_cols].dropna()
y = df.loc[X.index, "POP_NUM"]

# Add constant for intercept
X = sm.add_constant(X)

# Fit OLS regression
model = sm.OLS(y, X).fit()

# Save regression results
with open("./output/030/regression_analysis.txt", "w") as f:
    f.write("=== Regression Analysis: Population Structure ===\n\n")
    f.write("Dependent Variable: Population (numeric encoding)\n")
    f.write("Independent Variables: PC1-PC10\n\n")
    f.write(model.summary().as_text())
    f.write("\n\n=== Population Encoding ===\n")
    for i, pop in enumerate(le.classes_):
        f.write(f"{pop}: {i}\n")
    f.write("\n\n=== Variance Explained by PCs ===\n")
    for i in range(10):
        f.write(f"PC{i+1}: {var_explained[i]:.3f}%\n")
    f.write(f"\nCumulative variance (PC1-PC10): {sum(var_explained[:10]):.3f}%\n")

# Additional analysis: Geographic clustering if geographic info available
# For now, we'll create a distance matrix based on PC scores
from scipy.spatial.distance import pdist, squareform
from scipy.cluster.hierarchy import dendrogram, linkage

# Calculate mean PC values for each population
pop_means = df.groupby("POP")[pc_cols[:5]].mean()

# Calculate pairwise distances
dist_matrix = pdist(pop_means.values, metric="euclidean")
linkage_matrix = linkage(dist_matrix, method="ward")

# Create dendrogram
plt.figure(figsize=(12, 8))
dendrogram(linkage_matrix, labels=pop_means.index.tolist(), 
           color_threshold=0, above_threshold_color="black")
plt.title("Hierarchical Clustering of Populations based on PC1-PC5")
plt.xlabel("Population")
plt.ylabel("Euclidean Distance")
plt.xticks(rotation=45)
plt.tight_layout()
plt.savefig("./output/030/population_dendrogram.png", dpi=300)
plt.close()

# Save additional statistics
with open("./output/030/pca_statistics.txt", "w") as f:
    f.write("=== PCA Statistics Summary ===\n\n")
    f.write(f"Total number of individuals: {len(df)}\n")
    f.write(f"Number of populations: {len(POPULATIONS)}\n\n")
    f.write("Sample sizes per population:\n")
    pop_counts = df["POP"].value_counts().sort_index()
    for pop, count in pop_counts.items():
        f.write(f"{pop}: {count}\n")
    f.write("\n")
    f.write("Mean PC values per population (PC1-PC5):\n")
    f.write(pop_means.to_string())

print("PCA and regression analysis completed successfully!")
print(f"- PCA plot saved to: ./output/030/pca.png")
print(f"- Regression results saved to: ./output/030/regression_analysis.txt")
print(f"- Additional plots and statistics saved")
